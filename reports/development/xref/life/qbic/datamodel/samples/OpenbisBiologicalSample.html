18  package life.qbic.datamodel.samples;
19  
20  import java.util.List;
21  import java.util.Map;
22  
23  import life.qbic.xml.properties.Property;
24  
25  /**
26   * Class representing a sample created in a sample extraction experiment and from which test samples
27   * may be prepared
28   * 
29   * @author Andreas Friedrich
30   * 
31   */
32  public class OpenbisBiologicalSample extends AOpenbisSample {
33  
34    String primaryTissue;
35    String tissueDetailed;
36  
37    /**
38     * Create a new Biological Sample
39     * 
40     * @param openbisName Code of the sample
41     * @param experiment Experiment the sample is attached to
42     * @param secondaryName Secondary name of the sample (e.g. humanly readable identifier)
43     * @param additionalNotes Free text notes for the sample
44     * @param factors A list of conditions for the sample
45     * @param primaryTissue The primary tissue of this biological sample
46     * @param tissueDetailed Detailed tissue information
47     * @param parent Entity parent this sample was extracted from
48     */
49    public OpenbisBiologicalSample(String openbisName, String space, String experiment,
50        String secondaryName, String additionalNotes, List<Property> factors, String primaryTissue,
51        String tissueDetailed, String parent, String extID) {
52      super(openbisName, space, experiment, secondaryName, additionalNotes, factors, parent, extID,
53          "Q_BIOLOGICAL_SAMPLE");
54      this.primaryTissue = primaryTissue;
55      this.tissueDetailed = tissueDetailed;
56    }
57  
58    public OpenbisBiologicalSample(int tempID, List<AOpenbisSample> parents, String primaryTissue,
59        String tissueDetailed, String secondaryName, String externalID, List<Property> newFactors,
60        String additionalNotes) {
61      super(tempID, parents, "Q_BIOLOGICAL_SAMPLE", secondaryName, externalID, newFactors, additionalNotes);
62      this.primaryTissue = primaryTissue;
63      this.tissueDetailed = tissueDetailed;
64    }
65  
66    public Map<String, String> getValueMap() {
67      Map<String, String> res = super.getValueMap();
68      res.put("Q_PRIMARY_TISSUE", primaryTissue);
69      res.put("Q_TISSUE_DETAILED", tissueDetailed);
70      return res;
71    }
72  }
