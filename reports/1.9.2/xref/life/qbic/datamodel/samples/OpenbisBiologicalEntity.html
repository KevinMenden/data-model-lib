16  package life.qbic.datamodel.samples;
17  
18  import java.util.List;
19  import java.util.Map;
20  
21  import life.qbic.xml.properties.Property;
22  
23  /**
24   * Class representing a biological entity that will be used in an experiment and will be the root of
25   * the sample hierarchy
26   * 
27   * @author Andreas Friedrich
28   *
29   */
30  public class OpenbisBiologicalEntity extends AOpenbisSample {
31  
32    private String ncbiOrganism;
33    private String speciesInfo;
34  
35    /**
36     * Create a new Biological Entity
37     * 
38     * @param openbisName Code of the sample
39     * @param experiment Experiment the sample is attached to
40     * @param secondaryName Secondary name of the sample (e.g. humanly readable identifier)
41     * @param additionalNotes Free text notes for the sample
42     * @param factors A list of conditions of this sample
43     * @param ncbiOrganism The organism the entity belongs to
44     * @param specisInfo More information about the species
45     * @param extID External database identifier
46     */
47    public OpenbisBiologicalEntity(String openbisName, String space, String experiment,
48        String secondaryName, String additionalNotes, List<Property> factors, String ncbiOrganism,
49        String speciesInfo, String extID) {
50      super(openbisName, space, experiment, secondaryName, additionalNotes, factors, "", extID,
51          "Q_BIOLOGICAL_ENTITY");
52      this.ncbiOrganism = ncbiOrganism;
53      this.speciesInfo = speciesInfo;
54    }
55  
56    public Map<String, String> getValueMap() {
57      Map<String, String> res = super.getValueMap();
58      res.put("Q_NCBI_ORGANISM", ncbiOrganism);
59      res.put("Q_ORGANISM_DETAILED", speciesInfo);
60      return res;
61    }
62  }
