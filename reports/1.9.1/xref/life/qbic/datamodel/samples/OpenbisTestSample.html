16  package life.qbic.datamodel.samples;
17  
18  import java.util.List;
19  import java.util.Map;
20  
21  import life.qbic.xml.properties.Property;
22  
23  /**
24   * Class representing a sample created in a sample preparation that will be used to measure data
25   * 
26   * @author Andreas Friedrich
27   *
28   */
29  public class OpenbisTestSample extends AOpenbisSample {
30  
31    String Q_SAMPLE_TYPE;
32  
33    /**
34     * Create a new Test Sample
35     * 
36     * @param openbisName Code of the sample
37     * @param experiment Experiment the sample is attached to
38     * @param secondaryName Secondary Name of the sample (e.g. humanly readable identifier)
39     * @param additionalNotes Free text notes for the sample
40     * @param factors A list of conditions of this sample
41     * @param sampleType Measurement type of this sample (e.g. protein)
42     * @param parent Extract parent of this sample
43     */
44    public OpenbisTestSample(String openbisName, String space, String experiment,
45        String secondaryName, String additionalNotes, List<Property> factors, String sampleType,
46        String parent, String extID) {
47      super(openbisName, space, experiment, secondaryName, additionalNotes, factors, parent, extID,
48          "Q_TEST_SAMPLE");
49      this.Q_SAMPLE_TYPE = sampleType;
50    }
51  
52    public OpenbisTestSample(int tempID, List<AOpenbisSample> parents, String sampleType,
53        String secondaryName, String externalID, List<Property> newFactors, String additionalNotes) {
54      super(tempID, parents, "Q_TEST_SAMPLE", secondaryName, externalID, newFactors, additionalNotes);
55      this.Q_SAMPLE_TYPE = sampleType;
56    }
57  
58    public Map<String, String> getValueMap() {
59      Map<String, String> res = super.getValueMap();
60      res.put("Q_SAMPLE_TYPE", Q_SAMPLE_TYPE);
61      return res;
62    }
63  
64  }
